Je kunt bijvoorbeeld een klasse als volgt exporteren:

const numberRegexp = /^[0-9]{4}[A-Za-z]{2}$/

export class PostcodeChecker {
  isAcceptable(s: string) {
    s = s.replace(/\s/g, '')
    return s.length === 6 && numberRegexp.test(s)
  }
}
