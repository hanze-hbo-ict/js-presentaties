Destructuring arrays

Als een functie een array teruggeeft, gebruiken we in de regel een tmp-variabele om de waarden eruit te halen:

function foo() {
  return [3.14, 2.18, 42]
}

let tmp=foo();
a = tmp[0], b = tmp[1], c = tmp[2];

In de nieuwe versie van JavaScript kun je hiervoor array en object destructuring gebruiken:

let [a, b, c] = foo();

console.log(b); // 2.18
console.log(a); // 3.14
